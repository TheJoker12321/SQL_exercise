1. SELECT productVendor, COUNT(*) 
   FROM products
2. SELECT status, COUNT(*)
   FROM orders
   GROUP BY status
3. SELECT salesRepEmployeeNumber,  AVG(creditLimit)
   FROM customers
   GROUP BY salesRepEmployeeNumber
4. SELECT country, MIN(creditLimit), MAX(creditLimit)
   FROM customers
   GROUP BY country
5. SELECT officeCode, COUNT(employeeNumber)
   FROM employees
   GROUP BY officeCode
6. SELECT customerNumber, AVG(amount) AS average
   FROM payments
   GROUP BY customerNumber
   ORDER BY average DESC
7. SELECT productVendor, AVG(buyPrice)
   FROM products
   GROUP BY productVendor
8. SELECT salesRepEmployeeNumber, COUNT(customerNumber)
   FROM customers
   GROUP BY salesRepEmployeeNumber
9. SELECT orderDate, COUNT(orderNumber) AS ord_count
   FROM orders 
   GROUP BY orderDate
   ORDER BY ord_count DESC 
   LIMIT 1
10. SELECT productScale, COUNT(productCode)
    FROM products
    GROUP BY productScale
